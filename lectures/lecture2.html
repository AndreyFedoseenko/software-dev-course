Java

double d2 = a / 2.0; 
boolean t = d2 == 2.5 // true

Javascript

							
var a = 3 + 2 // a == 5
var b = a / 2 // b == 2.5
var t1 = a == '5' // true
var t2 = a === '5' // false

Функции

Java

							
int sum(int... args) {
    int result = 0;
    for (int arg : args) {
        result = result + arg;
    }
    return result;
};

int i1 = sum(1, 2, 3, 4); // i1 == 10
int i2 = sum(); // i2 == ?

Javascript

							
function sum(args) {
    var result = 0
    for (var arg in args)
        result = result + arg
     return result;
}

var i1 = sum([1, 2, 3, 4]) // i1 = 10
var i2 = sum() // i2 == ?

Рекурсия

Java

							
int factorial(int n) {
    if (n == 0 || n == 1 || n == 2) 
        return n;
    else
        return n * factorial(n - 1);
}

Javascript

							
function factorial(var n) {
    if (n == 0 || n == 1 || n == 2) 
        return n
    else 
        return n * factorial(n - 1)
}
